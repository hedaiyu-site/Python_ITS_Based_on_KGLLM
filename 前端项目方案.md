# Python ITS 前端项目方案

## 一、项目概述

### 1.1 项目名称
Python智能教学系统前端 (Python ITS Frontend)

### 1.2 项目描述
基于Vue 3框架开发的Python智能教学系统前端应用，为用户提供智能问答、知识图谱可视化、学习路径规划、代码辅助等功能的交互界面。

### 1.3 技术栈选型

| 类别 | 技术 | 版本 | 说明 |
|------|------|------|------|
| **核心框架** | Vue.js | 3.x | 渐进式JavaScript框架 |
| **构建工具** | Vite | 5.x | 下一代前端构建工具 |
| **状态管理** | Pinia | 2.x | Vue官方状态管理库 |
| **路由管理** | Vue Router | 4.x | Vue官方路由管理器 |
| **UI框架** | Element Plus | 2.x | Vue 3组件库 |
| **HTTP客户端** | Axios | 1.x | Promise based HTTP client |
| **图表可视化** | ECharts | 5.x | 数据可视化图表库 |
| **图谱可视化** | D3.js / vis-network | 7.x | 知识图谱可视化 |
| **代码编辑器** | Monaco Editor | 0.x | VS Code编辑器核心 |
| **样式预处理** | SCSS | - | CSS预处理器 |
| **代码规范** | ESLint + Prettier | - | 代码检查和格式化 |

```javascript
// src/api/request.js
import axios from 'axios'
import { useUserStore } from '@/stores/user'
import { ElMessage } from 'element-plus'
import router from '@/router'

const request = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL || 'http://localhost:8000',
  timeout: 30000,
  headers: {
    'Content-Type': 'application/json'
  }
})

request.interceptors.request.use(
  (config) => {
    const userStore = useUserStore()
    if (userStore.token) {
      config.headers.Authorization = `Bearer ${userStore.token}`
    }
    return config
  },
  (error) => {
    return Promise.reject(error)
  }
)

request.interceptors.response.use(
  (response) => {
    const res = response.data
    if (res.code !== 200) {
      ElMessage.error(res.message || '请求失败')
      return Promise.reject(new Error(res.message || '请求失败'))
    }
    return res
  },
  (error) => {
    if (error.response?.status === 401) {
      const userStore = useUserStore()
      userStore.logout()
      router.push('/login')
      ElMessage.error('登录已过期，请重新登录')
    } else {
      ElMessage.error(error.message || '网络错误')
    }
    return Promise.reject(error)
  }
)

export default request
```

## 五、状态管理设计

### 5.1 用户状态 (user.js)

```javascript
// src/stores/user.js
import { defineStore } from 'pinia'
import { authApi } from '@/api/auth'
import { storage } from '@/utils/storage'

export const useUserStore = defineStore('user', {
  state: () => ({
    token: storage.get('token') || '',
    userInfo: storage.get('userInfo') || null,
    isLoggedIn: !!storage.get('token')
  }),
  
  getters: {
    userId: (state) => state.userInfo?.user_id,
    username: (state) => state.userInfo?.username,
    level: (state) => state.userInfo?.level || 'beginner'
  },
  
  actions: {
    async login(credentials) {
      const res = await authApi.login(credentials)
      this.token = res.data.token
      this.userInfo = {
        user_id: res.data.user_id,
        username: res.data.username,
        email: res.data.email
      }
      this.isLoggedIn = true
      storage.set('token', this.token)
      storage.set('userInfo', this.userInfo)
      return res
    },
    
    async register(data) {
      return await authApi.register(data)
    },
    
    async fetchProfile() {
      const res = await authApi.getProfile(this.userId)
      this.userInfo = { ...this.userInfo, ...res.data }
      storage.set('userInfo', this.userInfo)
      return res
    },
    
    async updateProfile(data) {
      const res = await authApi.updateProfile({ user_id: this.userId, ...data })
      this.userInfo = { ...this.userInfo, ...res.data }
      storage.set('userInfo', this.userInfo)
      return res
    },
    
    logout() {
      this.token = ''
      this.userInfo = null
      this.isLoggedIn = false
      storage.remove('token')
      storage.remove('userInfo')
    }
  }
})
```

### 5.2 对话状态 (chat.js)

```javascript
// src/stores/chat.js
import { defineStore } from 'pinia'
import { chatApi } from '@/api/chat'
import { useUserStore } from './user'

export const useChatStore = defineStore('chat', {
  state: () => ({
    messages: [],
    history: [],
    currentContext: [],
    isLoading: false
  }),
  
  actions: {
    async askQuestion(question) {
      const userStore = useUserStore()
      this.isLoading = true
      this.messages.push({ role: 'user', content: question })
      
      try {
        const res = await chatApi.ask({
          user_id: userStore.userId,
          question,
          context: this.currentContext
        })
        
        this.messages.push({
          role: 'assistant',
          content: res.data.answer,
          responseId: res.data.response_id,
          sources: res.data.knowledge_sources
        })
        
        this.currentContext.push(
          { role: 'user', content: question },
          { role: 'assistant', content: res.data.answer }
        )
        
        return res
      } finally {
        this.isLoading = false
      }
    },
    
    async loadHistory(limit = 20, offset = 0) {
      const userStore = useUserStore()
      const res = await chatApi.getHistory({
        user_id: userStore.userId,
        limit,
        offset
      })
      this.history = res.data.history
      return res
    },
    
    async submitFeedback(responseId, rating, comment) {
      return await chatApi.submitFeedback({ response_id: responseId, rating, comment })
    },
    
    async clearHistory() {
      const userStore = useUserStore()
      await chatApi.clearHistory(userStore.userId)
      this.messages = []
      this.currentContext = []
      this.history = []
    }
  }
})
```

### 6.2 路由守卫

```javascript
// src/router/guards.js
import { useUserStore } from '@/stores/user'

export function setupRouterGuards(router) {
  router.beforeEach((to, from, next) => {
    const userStore = useUserStore()
    const requiresAuth = to.meta.requiresAuth
    
    document.title = to.meta.title ? `${to.meta.title} - Python ITS` : 'Python ITS'
    
    if (requiresAuth && !userStore.isLoggedIn) {
      next({ name: 'Login', query: { redirect: to.fullPath } })
    } else if (!requiresAuth && userStore.isLoggedIn && (to.name === 'Login' || to.name === 'Register')) {
      next({ name: 'Home' })
    } else {
      next()
    }
  })
}
```
